\subsection{Coordinate System}

\begin{equation}
\label{eq:cern:eta}
\eta = - \ln \tan \frac{\theta}{2}
\end{equation}

\begin{equation}
\label{eq:cern:y}
y = \frac{1}{2} \ln \frac{E + p_z}{E - p_z}
\end{equation}

\begin{equation}
\label{eq:cern:pt}
p_T = \sqrt{p_x^2 + p_y^2}
\end{equation}

\begin{equation}
\label{eq:cern:pz}
p_z = p_T \,\sinh \eta
\end{equation}

\begin{equation}
\label{eq:cern:dR}
\Delta R = \sqrt{ \Delta \phi^2 + \Delta \eta^2  }
\end{equation}
